$( function (){
     $( 'pre code' ).each( function (){
         texts = $(this).text(). replace (/&(?!#?[a-zA-Z0-9]+;)/g, '&' )
         . replace (//g, '>' ). replace (/ '/g, ' '' ). replace (/ "/g, '" ')       
         var codetext = texts.split( '\n' );
         var code = '' ;
         $(codetext).each( function (v){
             code += '<li>' +codetext[v]+ '</li>' ;
         })       
         code = '<ol>' +code+ '</ol>' ;
         $(this).html(code);
     });
});
